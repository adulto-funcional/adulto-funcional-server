package org.adultofuncional.main.auth.domain.service;

import org.adultofuncional.main.auth.application.dto.LoginRequest;
import org.adultofuncional.main.auth.application.dto.RegisterRequest;
import org.adultofuncional.main.auth.application.dto.AuthResponse;
import org.adultofuncional.main.shared.exception.BusinessException;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.stereotype.Service;

import java.util.UUID;

/**
 * ============================================================
 * AuthDomainService
 * ============================================================
 *
 * ¿Qué es?
 * Es un servicio encargado de manejar la lógica de autenticación,
 * incluyendo el registro y el inicio de sesión de usuarios.
 *
 * ¿Para qué sirve?
 * Permite centralizar todas las operaciones relacionadas con la
 * autenticación, garantizando consistencia en la validación de
 * datos y manejo de credenciales.
 *
 * ¿Qué hace?
 * - Gestiona el registro de usuarios
 * - Gestiona el inicio de sesión
 * - Valida los datos de entrada
 * - Encripta contraseñas
 * - Genera tokens de autenticación (simulados)
 *
 * Nota:
 * Esta es una versión simplificada que combina responsabilidades
 * de dominio y aplicación. En una arquitectura limpia, esto se
 * separaría en diferentes capas (Domain + Use Cases).
 *
 * ============================================================
 */
@Service
public class AuthDomainService {

    private final PasswordEncoder passwordEncoder;

    public AuthDomainService(PasswordEncoder passwordEncoder) {
        this.passwordEncoder = passwordEncoder;
    }

    /**
     * Maneja el registro de usuario
     */
    public AuthResponse register(RegisterRequest request) {

        validateRegister(request);

        String encodedPassword = passwordEncoder.encode(request.getPassword());
        String userId = UUID.randomUUID().toString();

        // Futuro: Guardar usuario en base de datos usando repository

        return AuthResponse.builder()
                .userId(userId)
                .email(request.getEmail())
                .token(generateFakeToken(request.getEmail()))
                .build();
    }

    /**
     * Maneja el inicio de sesión
     */
    public AuthResponse login(LoginRequest request) {

        validateLogin(request);

        // Simulación temporal (reemplazar por BD)
        String storedPassword = "$2a$10$exampleEncodedPassword";

        if (!passwordEncoder.matches(request.getPassword(), storedPassword)) {
            throw new BusinessException("Credenciales inválidas");
        }

        return AuthResponse.builder()
                .userId("usuario-mock-id")
                .email(request.getEmail())
                .token(generateFakeToken(request.getEmail()))
                .build();
    }

    /**
     * Validaciones para registro
     */
    private void validateRegister(RegisterRequest request) {

        if (request.getEmail() == null || request.getEmail().isBlank()) {
            throw new BusinessException("El correo es obligatorio");
        }

        if (request.getPassword() == null || request.getPassword().length() < 6) {
            throw new BusinessException("La contraseña debe tener al menos 6 caracteres");
        }
    }

    /**
     * Validaciones para login
     */
    private void validateLogin(LoginRequest request) {

        if (request.getEmail() == null || request.getEmail().isBlank()) {
            throw new BusinessException("El correo es obligatorio");
        }

        if (request.getPassword() == null || request.getPassword().isBlank()) {
            throw new BusinessException("La contraseña es obligatoria");
        }
    }

    /**
     * Generador de token temporal (luego reemplazar por JWT)
     */
    private String generateFakeToken(String email) {
        return "token-" + email + "-" + UUID.randomUUID();
    }
}
